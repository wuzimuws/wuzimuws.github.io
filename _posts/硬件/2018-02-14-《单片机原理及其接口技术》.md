循环led程序：

```
 #include "reg52.h"
 #include "intrins.h"
typedef unsigned char u8;
typedef unsigned int u16;
#define led P2

 sbit led1=P2^0;   //p20
 sbit led2=P2^1;

// 延时函数
 void delay(u16 i)
{
	while(i--);
}

 void main()
 {
 	u8 i=0;
	led=0xfe; //1111 1110	  0x
	delay(50000);
	while(1)
 	{
		for(i=0;i<7;i++)
		{
			led=_crol_(led,2);
			delay(50000);
				
		}

	}
 }
```

#### 蜂鸣器的设置

* 音调 改变输出波形的频率
* 声音大小 改变电平的占空比 高电平时间/整个周期的时间

有源蜂鸣器：低电平 发声，高电平 停止

![](http://p1trollkt.bkt.clouddn.com/15200758477189.jpg)

1.5～2.5kHZ
无源蜂鸣器：一定频率，有电路板

ULN2013芯片：大电流驱动
内有反相器，输入高电平，输出为低电平

![](http://p1trollkt.bkt.clouddn.com/15200763236857.jpg)


``` 
#include "reg52.h"
 #include "intrins.h"
typedef unsigned char u8;
typedef unsigned int u16;
#define led P2

 sbit beep=P1^5;

 void delay(u16 i)
{
	while(i--);
}

 u8 main()
 {
 	u8 i=0;
	for(i=0;i<100;i++)
	{
		beep=~beep;  //取反
		delay(100);
	}
	delay(5000);

	for(i=0;i<100;i++)  //设置另一个间隔不同的循环
	{
		beep=~beep;  
		delay(100);
	}
	delay(5000);
	delay(5000);
	delay(5000);
	return 0;    //若加此行，则运行一次即结束，否则一直循环
 }
```

```
 #include "reg52.h"
typedef unsigned char u8;

u8 code smgduan[]={0x3f,0x06,0x5b,0x4f,0x66,0x6d,0x7d,0x07,0x7f,0x6f,0x77,0x7c,0x39,0x5e,0x79,0x71,0x00};
```

```
	u8 i;

					LSA=1;LSB=0;LSC=0;break;
			case(2):
					LSA=0;LSB=1;LSC=0;break;
```

#### 独立按键
P1 P2 P3 准双向I/O口
P0漏极开路，所以外部需要上拉电阻，才能对高低电平进行控制

![](http://p1trollkt.bkt.clouddn.com/15205106379469.jpg)

通过软件消抖（通常）：10ms
或硬件消抖（成本高）：
![](http://p1trollkt.bkt.clouddn.com/15205110296339.jpg)



```
#include "reg52.h"
typedef unsigned char u8;
typedef unsigned int u16;

sbit led=P2^0;   //p20
sbit k1=P3^1;   //定义按键端口

// 延时函数
 void delay(u16 i)
{
	while(i--);
}

void keypress()
{
	if(k1==0)
	{
		delay(1000);  //约10ms 消抖
		if(k1==0)
		{
			led=~led;
		}
		while(!k1);   //松按键
	}
}

 void main()
 {
 	while(1)
 	{
		keypress();
	}
 }
```

```
 #include "reg52.h"
typedef unsigned char u8;
typedef unsigned int u16;

sbit LSA=P2^2;
sbit LSB=P2^3;
sbit LSC=P2^4;

#define GPIO_KEY P1
#define GPIO_DIG P0
u8 code smgduan[]={0x3f,0x06,0x5b,0x4f,0x66,0x6d,0x7d,0x07,0x7f,0x6f,0x77,0x7c,0x39,0x5e,0x79,0x71,0x00};
int input_dect = 0;
u8 KeyValue = 0;
u8 Key_Value_Save[8] = {0};
u8 i;


void delay(u16 i)
{
	while(i--);
}

void keyscan()  //矩阵按键扫描
{
	u8 a=0;
	GPIO_KEY=0x0f;
	if(GPIO_KEY!=0x0f)
	{	   
		delay(1000);
		if(GPIO_KEY!=0x0f)
		{
			GPIO_KEY=0x0f;
			switch(GPIO_KEY)
			{
				case(0x07): KeyValue=0; break;
				case(0x0b): KeyValue=1; break;
				case(0x0d): KeyValue=2; break;
				case(0x0e): KeyValue=3; break;
			}
			GPIO_KEY=0xf0;
			switch(GPIO_KEY)
			{
				case(0x70): KeyValue=KeyValue; break;
				case(0xb0): KeyValue=KeyValue+4; break;
				case(0xd0): KeyValue=KeyValue+8; break;
				case(0xe0): KeyValue=KeyValue+12; break;
			}
			while((a<50)&&(GPIO_KEY!=0xf0))		  //按键超时退出
			{
				delay(1000);
				a++;
			}
	
		} 
	input_dect++;
	}
}

void  dydisplay(u8 num)  // dynamic动态显示
{

		switch(num)
		{
			case(1):
					LSA=0;LSB=0;LSC=0;GPIO_DIG=smgduan[Key_Value_Save[1]];break;
			case(2):
					LSA=1;LSB=0;LSC=0;GPIO_DIG=smgduan[Key_Value_Save[2]];break;
			case(3):
					LSA=0;LSB=1;LSC=0;GPIO_DIG=smgduan[Key_Value_Save[3]];break;
			case(4):
					LSA=1;LSB=1;LSC=0;GPIO_DIG=smgduan[Key_Value_Save[4]];break;
			case(5):
					LSA=0;LSB=0;LSC=1;GPIO_DIG=smgduan[Key_Value_Save[5]];break;
			case(6):
					LSA=1;LSB=0;LSC=1;GPIO_DIG=smgduan[Key_Value_Save[6]];break;
			case(7):
					LSA=0;LSB=1;LSC=1;GPIO_DIG=smgduan[Key_Value_Save[7]];break;
			case(8):
					LSA=1;LSB=1;LSC=1;GPIO_DIG=smgduan[Key_Value_Save[8]];break;
		}
	/*	P0=smgduan[0];
			//延时 肉眼50ms
		P0=0x00;	//消隐*/
		delay(100);
}	
void main()
 {			
 	KeyValue = 16;		
	while(1)
	{
	keyscan();
	
	if(KeyValue<16)
	{
		Key_Value_Save[input_dect] = KeyValue;
		for (i = 0;i<=input_dect;i++)
		{
			dydisplay(i);
		}
	}
	}
 }
```
